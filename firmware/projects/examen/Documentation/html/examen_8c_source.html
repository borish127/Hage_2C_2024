162void distance_task(void *pvParameter)
163{
164 uint16_t distance;
165 uint8_t distance_case;
166 while(true){
167 distance=HcSr04ReadDistanceInCentimeters();
168 distance_case=distance_cases(distance);
169 switch (distance_case)
170 {
171 case 0:
172 green_light();
173 set_secure();
174 break;
175 case 1:
176 yellow_light();
177 set_caution();
178 send_caution();
179 break;
180 case 2:
181 red_light();
182 set_danger();
183 send_danger();
184 break;
185 }
186 }
187 vTaskDelay(READ_DISTANCE_DELAY/portTICK_PERIOD_MS);
188}

225void accelerometer_task(void *pvParameter)
226{
227 uint16_t acceleration_x;
228 uint16_t acceleration_y;
229 uint16_t acceleration_z;
230 uint16_t sum;
231 while(true){
232 AnalogInputReadSingle(CH0,&acceleration_x);
233 AnalogInputReadSingle(CH1,&acceleration_y);
234 AnalogInputReadSingle(CH2,&acceleration_z);
235 sum=1.65*(3+0.3*(acceleration_x+acceleration_y+acceleration_z)); // Convertido de voltaje a G
236 if(sum>4){
237 send_fall();
238 }
239 vTaskDelay(ACCELEROMETER_READ_DELAY/portTICK_PERIOD_MS);
240 }
241}
