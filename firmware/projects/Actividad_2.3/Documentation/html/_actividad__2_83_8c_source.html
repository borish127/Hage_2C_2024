158void distance_task(void *pvParameter)
159{
160 uint8_t distance;
161 while(true){ //Se crea un bucle que siempre corre cuando el sistema operativo lo asigna
162 ulTaskNotifyTake(pdTRUE,portMAX_DELAY);
163 if(active){
164 distance=read_distance(); //Lee la distancia
165 update_leds(create_intervals(distance)); //Enciende los leds correspondientes
166 if(!holding){
167 LcdItsE0803Write(distance); //Muestra la distancia en la pantalla
168 UartSendString(UART_PC,(const char*)UartItoa(distance,10));
169 UartSendString(UART_PC," cm\r\n");
170 }
171 }
172 }
173}
